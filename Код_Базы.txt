CREATE TABLE IF NOT EXISTS Artist (
id SERIAL PRIMARY KEY,
Artist VARCHAR(40),
Nickname VARCHAR(10) UNIQUE
);

CREATE TABLE IF NOT EXISTS Genres (
id SERIAL PRIMARY KEY,
Genre VARCHAR(10)
);

CREATE TABLE IF NOT EXISTS Alboms (
id SERIAL PRIMARY KEY,
Alboms VARCHAR(10),
Year integer check(Year>0) NOT NULL
);

CREATE TABLE IF NOT exists Artist_Alboms_Genres (
artist_id integer references Artist(id),
alboms integer references Alboms(id),
genres_id integer references Genres(id),
constraint pk primary key (Artist_id, Alboms_id,Genres_id)
);

CREATE TABLE IF NOT EXISTS Tracks (
id SERIAL PRIMARY KEY,
Tracks VARCHAR(10),
Duration integer check(Duration>0) NOT NULL
);
